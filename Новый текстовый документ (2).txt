#include <string>
#include <windows.h>
using namespace std;
class BankAccount
{
private:
	string owner;
	double balance;
public:
    BankAccount(string n, double count = 0)
        : owner(n),  balance(count)
    {
        if (balance < 0)


            cout << "Balance cant be < 0";
    }

    BankAccount()
        : BankAccount("Unkown",  0)
    {
    }
    bool deposit(double amount)
    {
        if (amount <= 0) {
            cout << "Невозможно внести отрицательную сумму!" << endl;
            return false;
        }
        else {
            balance += amount;
            cout << "Успешно внесено: " << amount << ". Новый баланс: " << balance << endl;
            return true;
        }
    }
    bool getMoney(double amount2)
    {
        //Функция снимает деньги со счета
        if (amount2 <= 0)
        {
            cout << "Невозможно внести отрицательную сумму!" << endl;
            return false;


        }
        else if(amount2<balance)
        {
            cout << "Баланс слишком маленький!" << endl;
            return false;


        }
        else
        {

            balance -= amount2;
            cout << "Успешно снято: " << amount2 << ". Новый баланс: " << balance << endl;
            return true;

        }



    }
    void input()
    {
        cout << "Введите имя владельца: ";
        getline(cin, owner);
        cout << "Введите начальный баланс: ";
        cin >> balance;
        cin.ignore(); 
        if (balance < 0) {
            cout << "Баланс не может быть меньше 0. Установлен в 0." << endl;
            balance = 0;
        }
    }
    void display() const
    {
        cout << "Владелец: " << owner << ", Баланс: " << balance << endl;
    }


};
int main()
{
    SetConsoleCP(1251);
    SetConsoleOutputCP(1251);
    BankAccount account;
    
    
    account.input();
    double amount;
    cout << "Введите сумму для внесения: ";
    cin >> amount;
    account.deposit(amount); 
    double amount2;
    cout << "Введите сумму для вычитания: ";
    cin >> amount2;
    account.getMoney(amount2);
}
